pragma solidity ^0.4.24;

contract ConcreteProduct 
{
    address DATABASE_CONTRACT;

    address lastBatch; 
    address product;
    
    uint256 id;
    uint256  QR;
    
    constructor(address _DATABASE_CONTRACT,address _lastBatch,address _product, uint256 _id,uint256 _qr) public
    {
        DATABASE_CONTRACT = _DATABASE_CONTRACT;
        lastBatch = _lastBatch;
        product = _product;
        id = _id;
        QR = _qr;
    }
    function getDetails() view public returns(address lastBatch, address product,  uint256 id, int256  QR)
    {
        return (lastBatch,product,id,QR);
    }
}
